<script>
  function calcularIMC() {
    const peso = parseFloat(document.getElementById('peso').value.replace(',', '.'))
    let alturaCm = parseFloat(document.getElementById('altura').value.replace(',', '.'))

    if (!peso || !alturaCm) return

    if (alturaCm < 3) alturaCm = alturaCm * 100

    if (alturaCm < 50 || alturaCm > 250) {
      alert('Altura inválida! Digite em cm (ex: 170) ou metros (ex: 1.70)')
      return
    }

    const alturaM = alturaCm / 100
    const imc = parseFloat((peso / (alturaM * alturaM)).toFixed(2))

    document.getElementById('imc').textContent = imc.toFixed(2)

    document.getElementById('abaixo').classList.remove('ativo')
    document.getElementById('ideal').classList.remove('ativo')
    document.getElementById('acima').classList.remove('ativo')

    if (imc < 18.5) document.getElementById('abaixo').classList.add('ativo')
    else if (imc < 25) document.getElementById('ideal').classList.add('ativo')
    else document.getElementById('acima').classList.add('ativo')

    document.getElementById('peso-hidden').value = peso
    document.getElementById('altura-hidden').value = alturaCm
    document.getElementById('acoes').style.display = 'flex'
  }

  function resetar() {
    document.getElementById('imc').textContent = '--'
    document.getElementById('abaixo').classList.remove('ativo')
    document.getElementById('ideal').classList.remove('ativo')
    document.getElementById('acima').classList.remove('ativo')
    document.getElementById('acoes').style.display = 'none'
    document.getElementById('peso').value = ''
    document.getElementById('altura').value = ''
  }
</script>
